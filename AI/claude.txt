ADSR in deinem Synthesizer
Aktuell hast du nur ein einfaches Decay (amplitude[k] *= DECAY_RATE). Für echtes ADSR brauchst du pro Taste einen Zustandsautomaten mit vier Phasen.

Konzept
Attack        Decay     Sustain   Release
  /\           \         ___        \
 /  \           \       |   |        \
/    \___________\______|   |_________\___
Taste gedrückt →              ← losgelassen
Implementierung
1. Structs & Parameter hinzufügen (global, vor setup()):

cpp
// ADSR-Parameter (Zeiten in Sekunden, Sustain 0.0–1.0)
float attackTime  = 0.01f;   // 10 ms
float decayTime   = 0.1f;    // 100 ms
float sustainLevel = 0.6f;   // 60% Lautstärke
float releaseTime = 0.3f;    // 300 ms

// Schrittgrößen pro Sample (werden aus den Zeiten berechnet)
float attackStep, decayStep, releaseStep;

enum ADSRState { IDLE, ATTACK, DECAY, SUSTAIN, RELEASE };
ADSRState adsrState[NUM_KEYS];
// amplitude[] bleibt dein Envelope-Wert (0.0 – 1.0)

void calcADSRSteps() {
  attackStep  = 1.0f / (attackTime  * SAMPLE_RATE);
  decayStep   = 1.0f / (decayTime   * SAMPLE_RATE);
  releaseStep = 1.0f / (releaseTime * SAMPLE_RATE);
}
2. In setup() initialisieren:

cpp
calcADSRSteps();
for (int i = 0; i < NUM_KEYS; i++) adsrState[i] = IDLE;
3. Den Sample-Loop ersetzen (den bisherigen amplitude-Block):

cpp
for (int k = 0; k < NUM_KEYS; k++) {
  bool pressed = keyPressed[k];

  switch (adsrState[k]) {
    case IDLE:
      if (pressed) {
        adsrState[k] = ATTACK;
        amplitude[k] = 0.0f;
      }
      break;

    case ATTACK:
      amplitude[k] += attackStep;
      if (amplitude[k] >= 1.0f) {
        amplitude[k] = 1.0f;
        adsrState[k] = DECAY;
      }
      if (!pressed) adsrState[k] = RELEASE;
      break;

    case DECAY:
      amplitude[k] -= decayStep;
      if (amplitude[k] <= sustainLevel) {
        amplitude[k] = sustainLevel;
        adsrState[k] = SUSTAIN;
      }
      if (!pressed) adsrState[k] = RELEASE;
      break;

    case SUSTAIN:
      amplitude[k] = sustainLevel;
      if (!pressed) adsrState[k] = RELEASE;
      break;

    case RELEASE:
      amplitude[k] -= releaseStep;
      if (amplitude[k] <= 0.0f) {
        amplitude[k] = 0.0f;
        adsrState[k] = IDLE;
      }
      if (pressed) {  // Retrigger
        adsrState[k] = ATTACK;
      }
      break;
  }
}
Diesen Block fügst du innerhalb der i-Schleife (pro Sample) ein, direkt vor dem mixedSample-Aufbau.

Wichtiger Hinweis zur Struktur
Da dein Loop blockweise (512 Samples) rechnet, aber keyPressed[] nur einmal pro Loop-Iteration abgefragt wird, ist die Granularität ~32 ms (512/16000). Das ist für ADSR grob, aber für ein Touch-Keyboard völlig ausreichend.

Wenn du feinere Hüllkurven willst, reduziere die Buffer-Größe auf z. B. 128 Samples.

Du könntest attackTime, decayTime etc. über die freien Taster oder Serial steuern und dann calcADSRSteps() erneut aufrufen – so bekommst du ein echtes spielbares Instrument.

Claude hat geantwortet: Das Problem liegt an der Architektur: Du berechnest ADSR-Zustände pro Sample innerhalb der 512er-Schleife, aber keyPressed[] wird außerhalb gelesen – das führt…
Das Problem liegt an der Architektur: Du berechnest ADSR-Zustände pro Sample innerhalb der 512er-Schleife, aber keyPressed[] wird außerhalb gelesen – das führt zu Inkonsistenzen und CPU-Last.

Hauptprobleme
1. ADSR pro Sample in der inneren Schleife → 512 × 14 State-Machine-Durchläufe pro Loop-Iteration = zu viel Overhead

2. keyPressed wird nur einmal gelesen, aber der Zustand ändert sich "mitten im Buffer" nicht → unnatürliche Hüllkurve

3. updateWaveTable() bei jedem Volume-Button → wird bei gedrücktem Button jeden Loop aufgerufen

Saubere Lösung: ADSR außerhalb der Sample-Schleife
Die Envelope wird pro Buffer-Block (nicht pro Sample) aktualisiert – bei 512 Samples / 16000 Hz = 32 ms Granularität, was für Touch völlig reicht.

cpp
// --- Globale ADSR-Structs ---
struct ADSR {
  float attack  = 0.01f;
  float decay   = 0.15f;
  float sustain = 0.6f;
  float release = 0.4f;
};

enum ADSRState : uint8_t { IDLE, ATTACK, DECAY, SUSTAIN, RELEASE };

ADSR adsr;
ADSRState adsrState[NUM_KEYS];
float envelope[NUM_KEYS];   // ersetzt amplitude[]
bool  wasPressed[NUM_KEYS]; // für Flankenerkennung

// --- Neue Funktion: Envelope pro Block aktualisieren ---
// bufferSize = Anzahl Samples im aktuellen Block
void updateEnvelopes(bool keyPressed[], int bufferSize) {
  float dt = (float)bufferSize / SAMPLE_RATE; // vergangene Zeit in Sekunden

  for (int k = 0; k < NUM_KEYS; k++) {
    bool pressed = keyPressed[k];
    bool justPressed  = pressed && !wasPressed[k];
    bool justReleased = !pressed && wasPressed[k];

    // Zustandsübergänge durch Flankenerkennung
    if (justPressed) {
      adsrState[k] = ATTACK;
    } else if (justReleased) {
      if (adsrState[k] != IDLE) adsrState[k] = RELEASE;
    }

    // Envelope-Wert berechnen
    switch (adsrState[k]) {
      case IDLE:
        envelope[k] = 0.0f;
        break;

      case ATTACK:
        if (adsr.attack > 0)
          envelope[k] += dt / adsr.attack;
        else
          envelope[k] = 1.0f;

        if (envelope[k] >= 1.0f) {
          envelope[k] = 1.0f;
          adsrState[k] = DECAY;
        }
        break;

      case DECAY:
        if (adsr.decay > 0)
          envelope[k] -= dt / adsr.decay * (1.0f - adsr.sustain);
        else
          envelope[k] = adsr.sustain;

        if (envelope[k] <= adsr.sustain) {
          envelope[k] = adsr.sustain;
          adsrState[k] = SUSTAIN;
        }
        break;

      case SUSTAIN:
        envelope[k] = adsr.sustain;
        break;

      case RELEASE:
        if (adsr.release > 0)
          envelope[k] -= dt / adsr.release;
        else
          envelope[k] = 0.0f;

        if (envelope[k] <= 0.0f) {
          envelope[k] = 0.0f;
          adsrState[k] = IDLE;
        }
        break;
    }

    wasPressed[k] = pressed;
  }
}

// --- Neuer loop() ---
void loop() {
  static int16_t buffer[512 * 2];
  size_t bytesWritten;
  bool keyPressed[NUM_KEYS] = {false};
  int activeKeys = 0;

  // Touch lesen
  for (int k = 0; k < NUM_KEYS; k++) {
    if ((touchRead(touchPins[k]) / 10) > thresholds[k]) {
      keyPressed[k] = true;
      activeKeys++;
    }
  }

  // Buttons – mit Entprellung per Flag
  static bool volUpHeld = false, volDnHeld = false;
  if (digitalRead(TASTER_VOL_UP) == HIGH && !volUpHeld) {
    volume = min(1.0f, volume + 0.005f);
    updateWaveTable();
    volUpHeld = true;
  } else if (digitalRead(TASTER_VOL_UP) == LOW) volUpHeld = false;

  if (digitalRead(TASTER_VOL_DN) == HIGH && !volDnHeld) {
    volume = max(0.0f, volume - 0.005f);
    updateWaveTable();
    volDnHeld = true;
  } else if (digitalRead(TASTER_VOL_DN) == LOW) volDnHeld = false;

  static bool octUpHeld = false, octDnHeld = false;
  if (digitalRead(TASTER_OCT_UP) == HIGH && !octUpHeld) {
    octshift = min(2, octshift + 1);
    octUpHeld = true;
  } else if (digitalRead(TASTER_OCT_UP) == LOW) octUpHeld = false;

  if (digitalRead(TASTER_OCT_DN) == HIGH && !octDnHeld) {
    octshift = max(-2, octshift - 1);
    octDnHeld = true;
  } else if (digitalRead(TASTER_OCT_DN) == LOW) octDnHeld = false;

  // Envelopes aktualisieren (einmal pro Buffer-Block)
  updateEnvelopes(keyPressed, 512);

  float octFactor = pow(2.0f, octshift);

  // Sample-Buffer füllen
  for (int i = 0; i < 512; i++) {
    int32_t mixedSample = 0;
    float   ampSum      = 0.0f;

    for (int k = 0; k < NUM_KEYS; k++) {
      if (envelope[k] > 0.001f) {
        float step = (basisNoten[k] * octFactor) * TABLE_SIZE / SAMPLE_RATE;
        phases[k] += step;
        if (phases[k] >= TABLE_SIZE) phases[k] -= TABLE_SIZE;

        mixedSample += (int32_t)(waveTable[(int)phases[k]] * envelope[k]);
        ampSum += envelope[k];
      }
    }

    if (ampSum > 1.0f) mixedSample = (int32_t)(mixedSample / ampSum);

    buffer[i * 2]     = (int16_t)constrain(mixedSample, -32767, 32767);
    buffer[i * 2 + 1] = buffer[i * 2];
  }

  i2s_write(I2S_NUM, buffer, sizeof(buffer), &bytesWritten, portMAX_DELAY);

  // LED
  if (activeKeys > 0)
    strip.setPixelColor(0, strip.Color(0, 155, 0));
  else
    strip.setPixelColor(0, strip.Color(0, 0, 155));
  strip.show();
}
